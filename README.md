# yo_polymer_per_install
bash install 
```
 cd
 wget https://npmjs.org/install.sh
 chmod +x install.sh
 sudo ./install.sh
 npm install gulp -g 
 npm install bower -g
 npm install -g yo
 wget https://iojs.org/dist/latest/iojs-v1.7.1-linux-x64.tar.gz
 tar xzf iojs-v1.7.1-linux-x64.tar.gz 
 cd iojs-v1.7.1-linux-x64/
 cp bin/* /usr/bin
 apt-get install ruby-full -y
 gem install compass
 ```
go to folder for app

```
 npm install -g generator-polymer --allow-root
 yo polymer
 npm install --allow-root
 bower install --allow-root

 ```
